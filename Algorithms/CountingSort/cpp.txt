#include <iostream>

using namespace std;

int max(int arr[], int n)
{
    int high;
    for (int i = 0 ; i < n ; i++){
        if (arr[i] > high)
            high = arr[i];
    }
    return high;
}


void countingSort(int *arr, int n)
{
    int maxNub = max(arr, n);

    int count[maxNub];
    for(int i; i < maxNub;i++){count[i] = 0;}
    int sort[n];
    for(int i; i < n;i++){sort[i] = 0;}


    for(int i = 0; i < n; i++){
  	    count[(arr[i])-1] += 1;
    }

    for(int i = 0; i < maxNub-1; i++){
      	count[i+1] += count[i];
    }
    
    for(int i = 0; i < n; i++){
        sort[count[arr[i]-1]-1] = arr[i];
      	count[arr[i]-1] -= 1;
    }
    
    for(int i = 0; i < n; i++){
        arr[i] = sort[i];
    }
    
}

int main()
{
    cout << "";
    int arr[5] = { 12, 34, 54, 2, 3};

    countingSort(arr, 5);
    
    for(int i = 0; i < 5; i++){
        cout << arr[i] << " ";
    }
    
    return 0;
}
